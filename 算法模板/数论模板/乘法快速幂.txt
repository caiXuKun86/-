#include<iostream>
using namespace std;
long long a,b,mod;
int power(int a,int b,int mod){
	long long ans=1;
	while(b>0){
		if((b&1)==1){
			ans=(ans*a)%mod;
		}
		a=(1ll*a*a)%mod;
		b>>=1;
	}
	return ans;
}
int main(){
	cin>>a>>b>>mod;
	int ans=power(a,b,mod);
	cout<<a<<"^"<<b<<" mod "<<mod<<"="<<ans<<endl;
}
