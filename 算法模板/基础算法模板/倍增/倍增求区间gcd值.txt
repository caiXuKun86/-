#include <bits/stdc++.h>
using namespace std;

int n,q;
int a[1005];
int lg[1005];
int st[ 1005 ][ 32];
int gcd(int a,int b){
	return b==0?a:gcd(b,a%b);
}
int main(){
	cin>>n>>q;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	
	lg[0]=-1;
	for(int i=1;i<=n;i++){
		lg[i]=lg[i/2]+1;
		st[i][0]=a[i];
		
	}
	for(int p=1;p<=lg[n];p++){
		for(int i=1;i+(1<<p)-1<=n;i++){
			st[i][p]=gcd(st[i][p-1],st[i+(1<<(p-1))][p-1]);
		}
	}
	
	for(int i=0,l,r;i<q;i++){
		cin>>l>>r;
		int p=lg[r-l+1];
		int ans=gcd(st[l][p],st[r-(1<<p)+1][p]);
		cout<<ans<<endl;
	}
	return 0;
}
