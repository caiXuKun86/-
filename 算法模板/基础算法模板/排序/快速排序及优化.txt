#include<iostream>
using namespace std;
#include<vector>
#include <iostream>
#include <cstdlib>
#include <ctime>


vector<int>v={2,1,3,5,4};
const int n=5;


int patition(int l,int r,int x){
	int a=l,xi=0;
	for(int i=l;i<=r;i++){
		if(v[i]<=x){
			swap(v[i],v[a]);
			if(v[a]==x){
				xi=a;
			}
			a++;
		}
		
	}
	swap(v[xi],v[a-1]);
	return a-1;
	
}

void quickSort(int l,int r){
	if(l>=r)return;
	int x=v[l+rand()%(r-l+1)];
	int m=patition(l,r,x);
	quickSort(l,m);
	quickSort(m+1,r);
	
}


//patition优化 
int first,last;
void patition2(int l,int r,int x){
	first=l,last=r;
	
	int i=l;
	while(i<=last){
		if(v[i]==x){
			i++;
		}else if(v[i]<x){
			swap(v[i++],v[first++]);
		}else{
			swap(v[i],v[last--]);
		}
	}
}
void quickSort2(int l,int r){
	if(l>=r)return;
	int x=v[l+rand()%(r-l+1)];
	patition2(l,r,x);
	int a=first,b=last;
	
	quickSort2(l,a-1);
	quickSort2(b+1,r);
	
}
