#include<iostream>
using namespace std;
#include<vector>
vector<int>v={2,1,3,5,4};

const int n=5;
vector<int>help(n,0);

void merge(int l,int r,int m){
	int i=l;
	int j=m+1;
	int k=l;
	
	while(i<=m&&j<=r){
		help[k++]=(v[i]<=v[j]?v[i++]:v[j++]);
	}
	while(i<=m){
		help[k++]=v[i++];
	}
	while(j<=r){
		help[k++]=v[j++];
	}
	for(int a=l;a<=r;a++){
		v[a]=help[a];
	}
}
void mergeSort(int l,int r){
	if(l>=r)return;
	int mid=(l+r)/2;
	mergeSort(l,mid);
	mergeSort(mid+1,r);
	merge(l,r,mid);
	
}
int main(){
	mergeSort(0,4);
	for(auto i:v){
		cout<<i<<endl;
	}

}
