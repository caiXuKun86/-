#include<iostream>
using namespace std;
#include<algorithm>
#define BASE 10
int arr[]={125,321,4421,3341,521,123,666,995,445,7829};
int cnts[BASE];
int help[10];

int n=10;

void jiShuSort(int bits){
    for(int offset=1;bits>0;offset*=BASE,bits--){
    
        fill(cnts,cnts+BASE,0);

        for(int i=0;i<n;i++){
            cnts[arr[i]/offset%BASE]++;
        }
        for(int i=0;i<BASE;i++){
            cnts[i]=cnts[i]+cnts[i-1];
        }
        for(int i=n-1;i>=0;i--){
            help[--cnts[(arr[i]/offset)%BASE]]=arr[i];
        }
        for(int i=0;i<n;i++){
            arr[i]=help[i];
        }
    }
}
int main(){

    jiShuSort(4);
    for(int i:arr){
        cout<<i<<' ';
    }
}
