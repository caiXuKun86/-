
#include<iostream>
using namespace std;
#include<queue>
#include<vector>
const int N=100010;
int indegree[N];
int n,m;
priority_queue<int,vector<int>,greater<int> >q;

int main()
{
	cin>>n>>m;
	vector<vector<int> >g(n+1);
	for(int i=1,x,y;i<=m;i++){
		cin>>x>>y;
		g[x].push_back(y);
		indegree[y]++;
	}
	for(int i=1;i<=n;i++){
		if(indegree[i]==0){
			q.push(i);
		}
	}
	while(!q.empty()){
		int k=q.top();
		q.pop();
		cout<<k<<' ';
		for(int i:g[k]){
			if(--indegree[i]==0){
				q.push(i);
			}
		}
		
		
		
	}
	
	
	
}
