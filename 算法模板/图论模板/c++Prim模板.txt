#include<iostream>
using namespace std;
#include<algorithm>
#include<queue>
#include<vector>

const int M=200005;
const int N=5005;
struct edge{
	int to;
	int w;
	friend bool operator<(edge x,edge y){
		return x.w>y.w;
	}
};
int vis[N];
int n,m;
priority_queue<edge>q;

int main(){
	cin>>n>>m;
	vector<vector<edge> >g(n+1);
	for(int i=0,a,b,w;i<m;i++){
		cin>>a>>b>>w;
		g[a].push_back({b,w});
		g[b].push_back({a,w});
	}
	vis[1]=true;
	for(auto i:g[1]){
		q.push(i);
	}
	int ans=0;
	int size=0;
	while(!q.empty()){
		auto k=q.top();
		q.pop();
		int to=k.to;
		int w=k.w;
		if(!vis[to]){
			for(auto i:g[to]){
				q.push(i);
			}
			vis[to]=true;
			ans+=w;
			size++;
		}
	}
	if(size==n-1)cout<<ans;
	else cout<<"orz";
	
	
	
}
