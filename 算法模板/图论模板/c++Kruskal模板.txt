#include<iostream>
using namespace std;
#include<algorithm>

struct edge{
	int w,a,b;
};
const int N=5005;
const int M=200005;
edge edges[M];
int father[N];
int n,m;

int find(int x){
	if(x!=father[x]){
		father[x]=find(father[x]);
	}
	return father[x];
}
int isSameSet(int x,int y){
	return find(x)==find(y);
}
void mergeSet(int x,int y){
	father[find(x)]=father[y];
}

int main(){
	int ans=0;
	int size=0;
	cin>>n>>m;
	for(int i=0,a,b,w;i<m;i++){
		cin>>a>>b>>w;
		edges[i]={w,a,b};
	}
	for(int i=1;i<=n;i++)father[i]=i;
	sort(edges,edges+m,[](edge& x,edge& y){return x.w<y.w;});
	for(int i=0;i<m;i++){
		int a=edges[i].a;
		int b=edges[i].b;
		if(!isSameSet(a,b)){
			ans+=edges[i].w;
			size++;
			mergeSet(a,b);
