//c++建图
//链式前向星
int head[100];
int nex[100];
int to[100];
int count=1;
//加点
void add(int u,int v){
	nex[count]=head[u];
	to[count]=v;
	head[u]=count;
	count++;
}
//遍历
int main()
{
	for(int i=1;i<=5;i++){
		for(int ei=head[i];ei>0;ei=nex[ei]){
		        cout<<i<<" "<<to[ei]<<endl;
		}
	}
}
