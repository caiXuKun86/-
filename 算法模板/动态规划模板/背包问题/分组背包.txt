#include<iostream>
using namespace std;
#include<cstring>

const int N=110;
const int V=110;
const int S=110;

int dp[N][V];

int weight[S][N];
int volume[S][N];
int group[N];

int n,v;

int main(){
	cin>>n>>v;
	for(int i=1,s;i<=n;i++){
		cin>>s;
		group[i]=s;
		for(int j=1;j<=s;j++){
			cin>>volume[j][i]>>weight[j][i];
		}
	}
	
	for(int i=1;i<=n;i++){
		for(int j=0;j<=v;j++){
			dp[i][j]=dp[i-1][j];
			for(int k=1;k<=group[i];k++){
				if(j-volume[k][i]>=0){
					dp[i][j]=std::max(dp[i][j],dp[i-1][j-volume[k][i]]+weight[k][i]);
				}
			}
		}
	}
	cout<<dp[n][v];
	return 0;
}
