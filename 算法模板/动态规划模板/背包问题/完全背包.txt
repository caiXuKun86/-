#include<iostream>
using namespace std;
#include<cstring>

const int N=1010;
const int V=1010;

int dp[N][V];

int weight[N];
int volume[N];
int n,v;



int main1(){
	cin>>n>>v;
	for(int i=1;i<=n;i++){
		cin>>volume[i]>>weight[i];
	}
	
    for(int i=1;i<=n;i++){
    	for(int j=0;j<=v;j++){
    		dp[i][j]=dp[i-1][j];
    		if(j-volume[i]>=0){
    			dp[i][j]=max(dp[i][j],dp[i][j-volume[i]]+weight[i]);
			}
		}
	}
	cout<<dp[n][v];
	return 0;
}
int main(){
	int dp[V];
	memset(dp,0,sizeof dp);
	cin>>n>>v;
	for(int i=1;i<=n;i++){
		cin>>volume[i]>>weight[i];
	}
	
    for(int i=1;i<=n;i++){
    	for(int j=0;j<=V;j++){
    		if(j-volume[i]>=0){
    			dp[j]=max(dp[j],dp[j-volume[i]]+weight[i]);
			}
		}
	}
	cout<<dp[v];
	return 0;
	
}
