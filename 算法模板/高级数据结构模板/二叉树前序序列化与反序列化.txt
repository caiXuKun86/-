class TreeNode{
    char val;
    TreeNode left;
    TreeNode right;

    public TreeNode() {
    }

    public TreeNode(char val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}
public class Text {
    public static int count=0;
    //反序列化
    public static TreeNode f1(String s){
        char ch=s.charAt(count++);
        if(ch=='#') return null;
        TreeNode node=new TreeNode(ch,null,null);
        node.left=f1(s);
        node.right=f1(s);
        return node;
    }
    //序列化
    public static void f2(TreeNode root,StringBuilder sb){
        if(root==null) {
            sb.append('#');
            return;
        }
        sb.append(root.val);
        f2(root.left,sb);
        f2(root.right,sb);

    }
  
    public static void main(String[] args) {
        String s="12##3##";
        TreeNode root=f1(s);
        StringBuilder sb=new StringBuilder();
        f2(root,sb);
        System.out.println(sb.toString());
    } 
}
